```typescript
/**
 * VdWAgent — lightweight, ephemeral reasoning agent for brief gap exploration.
 *
 * Lifecycle: spawning → active → (reasoning steps) → terminated
 * Self-terminates after maxIterations steps OR when marked for termination.
 *
 * In Phase 6: stub implementation (no actual LLM calls).
 * In Phase 7+: would call LLM inference for real bridging queries.
 */
export class VdWAgent {
  readonly id: string;
  readonly params: VdWSpawnParams;
  #status: "spawning" | "active" | "terminated" = "spawning";
  #stepsExecuted: number = 0;
  #synthQueries: string[] = [];
  #entitiesAdded: string[] = [];
  #relationsAdded: Array<{ from: string; to: string; type: string }> = [];

  constructor(id: string, params: VdWSpawnParams) {
    this.id = id;
    this.params = params;
  }

  get status() {
    return this.#status;
  }
  get stepsExecuted() {
    return this.#stepsExecuted;
  }

  /** Activate the agent. */
  activate(): void {
    this.#status = "active";
  }

  /**
   * Execute one reasoning step (stub for Phase 6).
   * Returns true if agent should continue, false if done.
   */
  async executeStep(): Promise<boolean> {
    if (this.#status !== "active") return false;
    this.#stepsExecuted++;

    // Phase 6 stub: generate synthetic bridging query
    const query =
      `VdW probe: What bridges community ${this.params.communityA} ` +
      `to community ${this.params.communityB}? (step ${this.#stepsExecuted})`;
    this.#synthQueries.push(query);

    // Generate synthetic bridge entity
    if (this.#stepsExecuted === 1) {
      const entity = `vdw-bridge-${this.params.gapId}-${this.id}`;
      this.#entitiesAdded.push(entity);
      this.#relationsAdded.push({
        from: entity,
        to: `community-${this.params.communityA}`,
        type: "vdw-bridge",
      });
    }

    // Self-terminate if maxIterations reached
    if (this.#stepsExecuted >= this.params.maxIterations) {
      this.#status = "terminated";
      return false;
    }

    return true;
  }

  /** Force terminate the agent. */
  terminate(): void {
    this.#status = "terminated";
  }

  /** Get the results of this agent's exploration. */
  getResults(): {
    synthQueries: readonly string[];
    entitiesAdded: readonly string[];
    relationsAdded: ReadonlyArray<{ from: string; to: string; type: string }>;
    stepsExecuted: number;
    success: boolean;
  } {
    return {
      synthQueries: [...this.#synthQueries],
      entitiesAdded: [...this.#entitiesAdded],
      relationsAdded: [...this.#relationsAdded],
      stepsExecuted: this.#stepsExecuted,
      success: this.#stepsExecuted > 0,
    };
  }
}
```

**Key methods:**

1. `updateRegime(regime: string): void`
   - Store current regime classification (received from regime:classification events).
   - Called by the orchestrator wiring layer (ComposeRootModule).

2. `updateH1Dimension(h1Dimension: number): void`
   - Track H^1 hysteresis: if h1 >= threshold, increment h1AboveThresholdCount.
   - If h1 < threshold, reset h1AboveThresholdCount to 0.
   - This implements the 2-iteration hysteresis from 06-RESEARCH.md.

3. `async evaluateAndSpawn(gaps: ReadonlyArray<GapMetrics>, h1Dimension: number, iteration: number): Promise<VdWAgent[]>`
   - Main decision method. Called when an H^1 obstruction is detected.
   - **Step 1: Check hysteresis.** If h1AboveThresholdCount < h1HysteresisCount, return empty (H^1 not sustained).
   - **Step 2: Check regime gating.**
     - 'stable': return empty (no exploration needed).
     - 'nascent': limit to 1 agent with reduced budget (500 tokens).
     - 'transitioning' or 'critical': full spawn logic.
   - **Step 3: Compute spawn count.**
     - Base count: 1 agent per gap.
     - If regime === 'critical' and h1 >= 5: 2 agents per gap (high-confidence exploration).
     - Cap at maxConcurrentAgents total (including already-running agents).
   - **Step 4: Compute token budget.**
     - budget = max(500, Math.floor(5000 / h1Dimension))
     - Inverse scaling: higher obstruction → lower per-agent budget → more probes.
   - **Step 5: Filter gaps by cooldown.**
     - Skip gaps that were spawned within spawnCooldownIterations.
   - **Step 6: Create VdWAgent instances.**
     - For each eligible gap, create VdWAgent with params.
     - Activate agent.
     - Add to #agents list.
     - Emit 'orch:vdw-agent-spawned' event.
   - **Step 7: Return spawned agents.**

4. `async runAgents(): Promise<void>`
   - Execute all active agents to completion (bounded by maxIterations).
   - For each agent with status === 'active':
     - Call executeStep() in a loop until it returns false.
   - After each agent completes:
     - Emit 'orch:vdw-agent-complete' event with results.
     - Remove from #agents list.

5. `getActiveAgentCount(): number`
   - Returns number of currently active agents.

6. `getH1HysteresisStatus(): { count: number; threshold: number; sustained: boolean }`
   - Returns current H^1 hysteresis state (for testing/monitoring).

7. `cleanup(): void`
   - Terminate all active agents.
   - Clear agent list.

**Design constraints:**

- Imports from TNA: only GapMetrics type (same isolation level as ObstructionHandler).
- No imports from sheaf, lcm, or soc modules.
- Uses EventBus for all event emission (same pattern as ObstructionHandler).
- Agents are queued/serialized: runAgents() processes them one at a time to avoid race conditions on graph mutations (per research doc).
