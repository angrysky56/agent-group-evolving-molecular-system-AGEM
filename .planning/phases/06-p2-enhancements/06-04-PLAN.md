```typescript
getNodePosition(nodeId: string): { x: number; y: number } | undefined {
  const meta = this.#nodeMetadata.get(nodeId);
  if (meta?.x !== undefined && meta?.y !== undefined) {

  return undefined;
```

```typescript
export class LayoutComputer extends EventEmitter {
  readonly #cooccurrenceGraph: CooccurrenceGraph;
  readonly #layoutConfig: LayoutConfig;
  readonly #computerConfig: LayoutComputerConfig;

  #lastLayout: LayoutOutput | null = null;
  #lastComputeIteration: number = 0;
  #currentInterval: number;
  #hasComputedInitial: boolean = false;

  constructor(
    cooccurrenceGraph: CooccurrenceGraph,
    layoutConfig?: Partial<LayoutConfig>,
    computerConfig?: Partial<LayoutComputerConfig>
  ) {
    super();
    this.#cooccurrenceGraph = cooccurrenceGraph;
    this.#layoutConfig = { ...DEFAULT_LAYOUT_CONFIG, ...layoutConfig };
    this.#computerConfig = { ...DEFAULT_COMPUTER_CONFIG, ...computerConfig };
    this.#currentInterval = this.#computerConfig.defaultComputeInterval;
  }
```

**Key methods:**

1. `computeLayout(iterations?: number): LayoutOutput`
   - If graph order < minGraphOrder: return trivial layout.
   - Get the graphology graph instance.
   - **Seed positions deterministically** (if not already positioned):
     - For each node: assign x = hash(nodeId) % 100, y = hash(nodeId, 2) % 100.
     - This ensures deterministic starting positions.
   - Run ForceAtlas2:
     ```typescript
     forceAtlas2Layout.assign(graph, {
       iterations: iterations ?? this.#layoutConfig.iterations,
       settings: {
         barnesHutOptimize: this.#layoutConfig.barnesHutOptimize,
         barnesHutTheta: this.#layoutConfig.barnesHutTheta,
         linLogMode: this.#layoutConfig.linLogMode,
         gravity: this.#layoutConfig.gravity,
         slowDown: this.#layoutConfig.slowDown,
         edgeWeightInfluence: this.#layoutConfig.edgeWeightInfluence,
         scalingRatio: this.#layoutConfig.scalingRatio,
         strongGravityMode: this.#layoutConfig.strongGravityMode,
       },
     });
     ```
   - Read positions from graph node attributes (ForceAtlas2 assigns x, y to nodes).
   - Compute convergence energy: sum of squared position deltas from previous layout.
   - Cache positions back to CooccurrenceGraph via updateNodePosition().
   - Build LayoutOutput.
   - Store as #lastLayout.
   - Emit 'tna:layout-updated' event.
   - Return LayoutOutput.

2. `computeIfDue(iteration: number): boolean`
   - If iteration - lastComputeIteration < currentInterval: return false.
   - Determine iterations count:
     - If hasComputedInitial: use incrementalMultiplier \* default iterations (faster update).
     - Else: use full iterations (initial convergence).
   - Call computeLayout(iterations).
   - Set hasComputedInitial = true.
   - Update lastComputeIteration.
   - Return true.

3. `adjustInterval(regime: string): void`
   - 'transitioning' or 'critical' → urgentComputeInterval.
   - 'stable' → relaxedComputeInterval.
   - Otherwise → defaultComputeInterval.

4. `exportJSON(): LayoutExportJSON`
   - Build JSON export with node positions, community assignments, edges, metadata.
   - Include node: id, x, y, communityId, betweennessCentrality, label (lemma).
   - Include edges: source, target, weight.
   - Include metadata: nodeCount, edgeCount, energy, iterations, timestamp.
   - This is the primary output for D3.js/Sigma.js visualization consumers.

5. `getLastLayout(): LayoutOutput | null`
   - Return cached last layout.

6. `isConverged(): boolean`
   - Return true if lastLayout energy < convergenceEnergyThreshold.

7. `#computeEnergy(previousPositions: Map<string, NodePosition> | null, currentPositions: Map<string, NodePosition>): number`
   - If no previous positions: return Infinity (first computation).
   - Sum squared distances: Sigma((x_new - x_old)^2 + (y_new - y_old)^2) / nodeCount.
   - This measures how much the layout changed (convergence indicator).

8. `#seedPositions(graph: any): void`
   - For each node without existing x/y attributes:
     - Compute deterministic position from node ID hash.
     - Set x, y on the graphology node attributes.
   - Uses simple string hash: `hash(str) = str.charCodeAt(i) * 31^i`.

**Design constraints:**

- ZERO imports from lcm/, sheaf/, soc/, or orchestrator/.
- Uses createRequire for graphology-layout-forceatlas2 (same pattern as CentralityAnalyzer).
- EventEmitter for layout-updated events (same pattern as CentralityAnalyzer).
- Deterministic seed for reproducible layouts.
